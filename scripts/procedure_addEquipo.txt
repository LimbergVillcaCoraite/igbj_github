USE IGBJ;
DELIMITER //
CREATE PROCEDURE add_Equipo(
    IN _CODEQUIPO varchar(50),
    IN _NIT varchar(30),
    IN _MARCAEQUIPO varchar(20),
    IN _MODELOEQUIPO varchar(30),
    IN _SERIEEQUIPO varchar(30),
    IN _ANIOFABREQUIIPO date,
    IN _PROCEDENCIAEQUIPO varchar(30),
    IN _FECHAINGRESO date,
    IN _FOTOGRAFIA longblob,
    IN _FECHAREGISTRO date,
    IN _FECHAMODIFICACION date,
    IN _USUARIOREGISTRO date,
    IN _CODACTIVOFIJO varchar(15),
    IN _DESCRIPCION varchar(200)
)
BEGIN
        INSERT INTO EQUIPOS(CODEQUIPO,NIT,MARCAEQUIPO,MODELOEQUIPO,SERIEEQUIPO,ANIOFABREQUIIPO,PROCEDENCIAEQUIPO,FECHAINGRESO,FOTOGRAFIA,FECHAREGISTRO,FECHAMODIFICACION,USUARIOREGISTRO,CODACTIVOFIJO,DESCRIPCION) VALUES (_CODEQUIPO,_NIT,_MARCAEQUIPO,_MODELOEQUIPO,_SERIEEQUIPO,_ANIOFABREQUIIPO,_PROCEDENCIAEQUIPO,_FECHAINGRESO,_FOTOGRAFIA,_FECHAREGISTRO,_FECHAMODIFICACION,_USUARIOREGISTRO,_CODACTIVOFIJO,_DESCRIPCION);
END
//
