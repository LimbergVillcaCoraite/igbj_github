USE IGBJ;
DELIMITER //
CREATE PROCEDURE add_Persona
(
   IN _CI varchar(15),
   IN _NOMREUSUARIO varchar(12),
   IN _NOMBRES varchar(30),
   IN _APPATERNO varchar(50),
   IN _APMATERNO varchar(30),
   IN _FECHANAMIENTO date,
   IN _DIRECCION varchar(50),
   IN _TELEFONO varchar(8),
   IN _CORREO varchar(30),
   IN _CELULARPERSONA varchar(8),
   IN _SEXO char(1),
   IN _COMPLEMETOCI varchar(5),
   IN _EXPEDIDOCI varchar(5),
   IN _PROFESION varchar(50)
)
BEGIN
	INSERT INTO PERSONA(CI,NOMREUSUARIO,NOMBRES,APPATERNO,APMATERNO,FECHANAMIENTO,DIRECCION,TELEFONO,CORREO,CELULARPERSONA,SEXO,COMPLEMETOCI,EXPEDIDOCI,PROFESION) VALUES (_CI,_NOMREUSUARIO,_NOMBRES,_APPATERNO,_APMATERNO,_FECHANAMIENTO,_DIRECCION,_TELEFONO,_CORREO,_CELULARPERSONA,_SEXO,_COMPLEMETOCI,_EXPEDIDOCI,_PROFESION);
END
